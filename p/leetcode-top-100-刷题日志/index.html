class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> hashtable = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; ++i) {
            if (hashtable.containsKey(target - nums[i])) {
                return new int[]{hashtable.get(target - nums[i]), i};
            }
            hashtable.put(nums[i], i);
        }
        return new int[0];
    }

class Solution {
    public void moveZeroes(int[] nums) {
        int n = nums.length, left = 0, right = 0;
        while (right < n) {
            if (nums[right] != 0) {
                swap(nums, left, right);
                left++;
            }
            right++;
        }
    }

    public void swap(int[] nums, int left, int right) {
        int temp = nums[left];
        nums[left] = nums[right];
        nums[right] = temp;
    }
}

        // ÂàùÂßãÂåñ map

class Solution {
    public String minWindow(String S, String t) {
        char[] s = S.toCharArray();
        int m = s.length;
        int ansLeft = -1;
        int ansRight = m;
        int left = 0;
        int[] cntS = new int[128]; // s Â≠ê‰∏≤Â≠óÊØçÁöÑÂá∫Áé∞Ê¨°Êï∞
        int[] cntT = new int[128]; // t ‰∏≠Â≠óÊØçÁöÑÂá∫Áé∞Ê¨°Êï∞
        for (char c : t.toCharArray()) {
            cntT[c]++;
        }
        for (int right = 0; right < m; right++) { // ÁßªÂä®Â≠ê‰∏≤Âè≥Á´ØÁÇπ
            cntS[s[right]]++; // Âè≥Á´ØÁÇπÂ≠óÊØçÁßªÂÖ•Â≠ê‰∏≤
            while (isCovered(cntS, cntT)) { // Ê∂µÁõñ

                    ansLeft = left; // ËÆ∞ÂΩïÊ≠§Êó∂ÁöÑÂ∑¶Âè≥Á´ØÁÇπ

                cntS[s[left++]]--; // Â∑¶Á´ØÁÇπÂ≠óÊØçÁßªÂá∫Â≠ê‰∏≤
            }
        }
        return ansLeft < 0 ? "" : S.substring(ansLeft, ansRight + 1);
    }

    private boolean isCovered(int[] cntS, int[] cntT) {
        for (int i = 'A'; i <= 'Z'; i++) {
            if (cntS[i] < cntT[i]) {
                return false;
            }
        }
        for (int i = 'a'; i <= 'z'; i++) {
            if (cntS[i] < cntT[i]) {
                return false;
            }
        }
        return true;
    }
}
